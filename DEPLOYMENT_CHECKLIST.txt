API ENDPOINT:
────────────
@app.route('/api/qa/<category>')
def get_qa_by_category(category):
    questions = PreBuiltQA.query.filter_by(category=category)
    return jsonify([{
        'id': q.id,
        'question': q.question
    } for q in questions])


FALLBACK RESPONSE:
──────────────────
def get_aura_response(user_message):
    try:
        return call_gemini_api(user_message)
    except:
        # Fallback to pre-built Q&A
        for qa_pair in QA_PAIRS:
            if is_similar(user_message, qa_pair['question']):
                return qa_pair['response']
